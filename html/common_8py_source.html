14 def str_to_list(x):
15  # function to return a list based on a formatted string
16  result = []
17  if '[' in x:
18  result = ast.literal_eval(x)
19  else:
20  for part in x.split(','):
21  if ('-' in part):
22  a, b, c = part.split('-')
23  a, b, c = int(a), int(b), int(c)
24  result.extend(range(a, b, c))
25  elif ('txt' in part):
26  result.append(part)
27  else:
28  a = int(part)
29  result.append(a)
30  return result
